4``InteractionReward`` base for condition-triggered rewards, and the

17def _compute_fwd_positions(prev_state): 

18 """Compute forward positions, clipped indices, bounds mask, and forward type IDs.""" 

19 # Direction vector table: Right=0, Down=1, Left=2, Up=3 

20 dir_vec_table = xp.array([[0, 1], [1, 0], [0, -1], [-1, 0]], dtype=xp.int32) 

21 

22 fwd_pos = prev_state.agent_pos + dir_vec_table[prev_state.agent_dir] # (n_agents, 2) 

23 H, W = prev_state.object_type_map.shape 

24 fwd_r = xp.clip(fwd_pos[:, 0], 0, H - 1) 

25 fwd_c = xp.clip(fwd_pos[:, 1], 0, W - 1) 

26 in_bounds = ( 

27 (fwd_pos[:, 0] >= 0) & (fwd_pos[:, 0] < H) & (fwd_pos[:, 1] >= 0) & (fwd_pos[:, 1] < W) 

28 ) 

29 fwd_types = prev_state.object_type_map[fwd_r, fwd_c] # (n_agents,) 

30 

31 return fwd_pos, fwd_r, fwd_c, in_bounds, fwd_types

34class Reward: 

35 """Base class for reward functions. 

36 

37 Subclasses define compute() which receives StateView objects and returns 

38 (n_agents,) float32 reward arrays. The returned values are the final 

39 rewards -- apply any scaling or broadcasting inside compute(). 

40 

41 Parameters are passed via __init__ kwargs and stored in self.config. 

42 

43 Usage:: 

44 

45 class DeliveryReward(Reward): 

46 def compute(self, prev_state, state, actions, reward_config): 

47 coefficient = self.config.get("coefficient", 1.0) 

48 ... 

49 return rewards # (n_agents,) float32 

50 

51 

52 config = { 

53 "rewards": [DeliveryReward(coefficient=1.0, common_reward=True)], 

54 } 

55 """

75class InteractionReward(Reward): 

76 """Declarative base for condition-triggered rewards. 

77 

78 Class attributes (declare what triggers the reward): 

79 action: "pickup_drop", "toggle", or None (any/no action check). 

80 Subclasses MUST set this explicitly -- there is no default. 

81 holds: str type name agent must hold, or None 

82 faces: str type name agent must face (forward cell), or None 

83 overlaps: str type name agent must stand on, or None 

84 direction: int direction agent must face (0=R,1=D,2=L,3=U), or None 

85 

86 Instance config (runtime tuning via __init__ kwargs): 

87 coefficient: float scaling factor (default 1.0) 

88 common_reward: bool broadcast to all agents (default False) 

89 

90 Override extra_condition() for domain-specific checks beyond 

91 the standard conditions (pot capacity, timers, etc.). 

92 

93 Examples:: 

94 

95 class OnionInPotReward(InteractionReward): 

96 action = "pickup_drop" 

97 holds = "onion" 

98 faces = "pot" 

99 

100 

101 class GoalReward(InteractionReward): 

102 action = None 

103 overlaps = "goal" 

104 """ 

105 

106 action = _UNSET # subclasses MUST set to "pickup_drop", "toggle", or None 

107 holds = None # type name agent must hold 

108 faces = None # type name in forward cell 

109 overlaps = None # type name agent stands on 

110 direction = None # direction agent must face (0-3) 

111 

112 def extra_condition(self, mask, prev_state, fwd_r, fwd_c, reward_config): 

113 """Override to add conditions beyond the declarative attributes. 

114 

115 Return narrowed boolean mask. 

116 """ 

117 return mask

119 def compute(self, prev_state, state, actions, reward_config): 

120 """Compute (n_agents,) float32 rewards from declarative conditions.""" 

121 if self.action is _UNSET: 

122 raise TypeError( 

123 f"{type(self).__name__} must set action to 'pickup_drop', 'toggle', or None" 

124 ) 

125 

126 type_ids = reward_config["type_ids"] 

127 n_agents = reward_config["n_agents"] 

128 coefficient = self.config.get("coefficient", 1.0) 

129 common_reward = self.config.get("common_reward", False) 

130 

131 mask = xp.ones(n_agents, dtype=xp.bool_) 

132 

133 # Action check 

134 if self.action == "pickup_drop": 

135 mask = mask & (actions == reward_config["action_pickup_drop_idx"]) 

136 elif self.action == "toggle": 

137 mask = mask & (actions == reward_config["action_toggle_idx"]) 

138 # action=None -> no action filter 

139 

140 # Inventory check 

141 if self.holds is not None: 

142 mask = mask & (prev_state.agent_inv[:, 0] == type_ids[self.holds]) 

143 

144 # Direction check 

145 if self.direction is not None: 

146 mask = mask & (prev_state.agent_dir == self.direction) 

147 

148 # Forward-cell check (only compute fwd positions if needed) 

149 fwd_r = fwd_c = None 

150 if self.faces is not None: 

151 _, fwd_r, fwd_c, in_bounds, fwd_types = _compute_fwd_positions(prev_state) 

152 mask = mask & in_bounds & (fwd_types == type_ids[self.faces]) 

153 

154 # Overlap check (agent's current cell) 

155 if self.overlaps is not None: 

156 rows = prev_state.agent_pos[:, 0] 

157 cols = prev_state.agent_pos[:, 1] 

158 mask = mask & (prev_state.object_type_map[rows, cols] == type_ids[self.overlaps]) 

159 

160 # Extra condition hook 

161 mask = self.extra_condition(mask, prev_state, fwd_r, fwd_c, reward_config) 

162 

163 # Apply coefficient + broadcast 

164 if common_reward: 

165 n_earners = xp.sum(mask.astype(xp.float32)) 

166 return xp.full(n_agents, n_earners * coefficient, dtype=xp.float32) 

167 else: 

168 return mask.astype(xp.float32) * coefficient
